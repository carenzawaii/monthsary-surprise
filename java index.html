function showHearts() { const heartContainer = document.getElementById('heartContainer'); for (let i = 0; i < 50; i++) { const heart = document.createElement('div'); heart.className = 'heart'; heart.innerHTML = '❤️'; heart.style.left = `${Math.random() * 100}%`; heart.style.top = `${Math.random() * 100}%`; heartContainer.appendChild(heart); } }
